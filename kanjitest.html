const QUESTIONS=[


{q:'へん(じ)をする',a:'事'},{q:'(どう)物園',a:'動'},{q:'人(ぶつ)の顔',a:'物'},{q:'(の)りごごち',a:'乗'},{q:'(しゅ)人公',a:'主'},


{q:'(はし)のたもと',a:'橋'},{q:'川(ぎし)を歩く',a:'岸'},{q:'(はっ)見する',a:'発'},{q:'よ(そう)する',a:'予'},{q:'空(そう)する',a:'想'},


{q:'理(ゆう)を話す',a:'由'},{q:'体(おん)が高い',a:'温'},{q:'(かん)字を書く',a:'漢'},{q:'言葉の(い)味',a:'意'},{q:'記(ごう)を書く',a:'号'},


{q:'(と)い合わせ',a:'問'},{q:'(れん)しゅう',a:'練'},{q:'国語の学(しゅう)',a:'習'},{q:'心に(かん)じる',a:'感'},{q:'(うん)どう場',a:'運'},


{q:'回(てん)する',a:'転'},{q:'うめの(み)',a:'実'},{q:'(しょう)かする',a:'消'},{q:'へん(か)する',a:'化'},{q:'池の水(めん)',a:'面'},


{q:'(み)をまもる',a:'身'},{q:'(だい)一話',a:'第'},{q:'(つぎ)の日',a:'次'},{q:'魚が(し)ぬ',a:'死'},{q:'四十(ど)近い',a:'度'},


{q:'研(きゅう)する',a:'究'},{q:'十(びょう)くらい',a:'秒'},{q:'(しょう)わの日',a:'昭'},{q:'(わ)食の店',a:'和'},{q:'図書(かん)の本',a:'館'},


{q:'文(しょう)を書く',a:'章'},{q:'虫の体(ぜん)体',a:'全'},{q:'本の(だい)名',a:'題'},{q:'木の(かわ)',a:'皮'},{q:'(あい)手に話す',a:'相'},


{q:'目を(ひらく)',a:'開く'},{q:'ふり(かえる)',a:'返る'},{q:'山に(のぼる)',a:'登る'},{q:'本で(しらべる)',a:'調べる'},{q:'言葉で(あらわす)',a:'表す'},


{q:'(おもい)にもつ',a:'重い'},{q:'人を(あつめる)',a:'集める'},{q:'魚が(およぐ)',a:'泳ぐ'},{q:'きまりを(まもる)',a:'守る'},{q:'川に(ながす)',a:'流す'}


];


const LS='kanji50_v1';let hist=JSON.parse(localStorage.getItem(LS)||'{"total":0,"correct":0,"wrong":[]}');let order=[],idx=0,mode='write',session=[],lastMode='write';


function save(){localStorage.setItem(LS,JSON.stringify(hist));updateStats()}function shuffle(a){return [...a].sort(()=>Math.random()-.5)}function updateStats(){total.textContent=hist.total;correct.textContent=hist.correct;rate.textContent=hist.total?Math.round(hist.correct/hist.total*100)+'%':'0%';wrongCount.textContent=hist.wrong.length}


function ids(kind){let all=QUESTIONS.map((_,i)=>i);return kind==='wrong'?(hist.wrong.length?shuffle(hist.wrong):shuffle(all)):shuffle(all)}


function start(m,kind){mode=m;lastMode=m;order=ids(kind);idx=0;session=[];home.classList.add('hidden');result.classList.add('hidden');listPanel.classList.add('hidden');quiz.classList.remove('hidden');showQ()}


function showQ(){let n=order[idx],it=QUESTIONS[n];progress.textContent=(idx+1)+' / '+order.length+'　No.'+(n+1);bar.style.width=((idx)/order.length*100)+'%';q.textContent=it.q;answer.classList.add('hidden');judge.classList.add('hidden');nextBtn.classList.add('hidden');feedback.classList.add('hidden');writeBox.classList.toggle('hidden',mode!=='write');choiceBox.classList.toggle('hidden',mode!=='choice');if(mode==='choice')choices(n)}


function reveal(){answer.textContent=QUESTIONS[order[idx]].a;answer.classList.remove('hidden');judge.classList.remove('hidden')}


function mark(ok){record(ok);nextBtn.classList.remove('hidden');judge.classList.add('hidden')}


function choices(n){let correct=QUESTIONS[n].a;let pool=QUESTIONS.map(x=>x.a).filter(x=>x!==correct);let opts=shuffle([correct,...shuffle(pool).slice(0,2)]);choiceBox.innerHTML='';opts.forEach(o=>{let b=document.createElement('button');b.textContent=o;b.onclick=()=>{[...choiceBox.children].forEach(c=>c.disabled=true);if(o===correct){b.classList.add('correct');feedback.textContent='○ 正解！';record(true)}else{b.classList.add('wrong');[...choiceBox.children].find(c=>c.textContent===correct).classList.add('correct');feedback.textContent='答え：'+correct;record(false)}feedback.classList.remove('hidden');nextBtn.classList.remove('hidden')};choiceBox.appendChild(b)})}


function record(ok){let n=order[idx];hist.total++;if(ok){hist.correct++;hist.wrong=hist.wrong.filter(x=>x!==n)}else{if(!hist.wrong.includes(n))hist.wrong.push(n)}session.push({n,ok});save()}


function next(){idx++;if(idx>=order.length)finish();else showQ()}


function finish(){quiz.classList.add('hidden');result.classList.remove('hidden');bar.style.width='100%';let c=session.filter(x=>x.ok).length,w=session.length-c;rTotal.textContent=session.length;rCorrect.textContent=c;rRate.textContent=session.length?Math.round(c/session.length*100)+'%':'0%';rWrong.textContent=w;wrongList.innerHTML=w?session.filter(x=>!x.ok).map(x=>`<div>No.${x.n+1} ${QUESTIONS[x.n].q} → ${QUESTIONS[x.n].a}</div>`).join(''):'<div>全部正解です！</div>'}


function goHome(){quiz.classList.add('hidden');result.classList.add('hidden');listPanel.classList.add('hidden');home.classList.remove('hidden');updateStats()}


function resetHistory(){if(confirm('学習履歴をリセットしますか？')){hist={total:0,correct:0,wrong:[]};save()}}


function showList({home.classList.add('hidden');listPanel.classList.remove('hidden');allList.innerHTML=QUESTIONS.map((x,i)=>`<div>No.${i+1} ${x.q} → ${x.a}</div>`).join('')}


updateStats();
